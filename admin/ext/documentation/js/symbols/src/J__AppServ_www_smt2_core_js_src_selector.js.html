 28 /**
 29  * Pre-compiles an Selector query.
 30  * <p>When creating a <code>new</code> instance of the <code>Selector</code>, the query are 
 31  * pre-compiled for later execution with a call to {@link exec}.</p>
 32  * <p>The constructor can also be called as a function, without the <code>new</code> keyword. Then 
 33  * the query is executed against the optional context element, otherwise current document.</p>
 34  * <h4>Example:</h4>
 35  * <pre>
 36  *   // Constructor syntax:
 37  *   new Selector('div > p').exec(document).forEach(function (e) {
 38  *     e.style.color = 'red';
 39  *   });
 40  *   // Or,  shorthand syntax:
 41  *   Selector('div > p', document).forEach(function (e) {
 42  *     e.style.color = 'red';
 43  *   });
 44  * </pre>
 45  * @constructor Selector
 46  * @param {string} pattern - selector pattern.
 47  * @param {optional Object} context - document or context element.
 48  */
 49 function Selector (p, c) {
 50   if (!(this instanceof Selector)) return new Selector(p).exec(c);
 51   if (!qsa) this.exec = cache[p] || (cache[p] = new compile(p));
 52   this.pattern = p;
 53 }
 54 Selector.prototype = {
 55   constructor : Selector,
 56 /**
 57  * Execute a selector query.
 58  * <h4>Example:</h4>
 59  * <pre>
 60  *   new Selector('div > p').exec(document).forEach(function (e) {
 61  *     e.style.color = 'red';
 62  *   });
 63  * </pre>
 64  * @function {Array} exec
 65  * @param {optional Object} context - document or context element, otherwise current document.
 66  * @returns Non-live <code>Array</code> with matching elements.
 67  */
 68   exec : function (c) {
 69     var pe = this.patchElement, pa = this.patchArray, p = this.pattern, r = pe
 70       ? map.call((c||d).querySelectorAll(p), pe, this)
 71       : Array.prototype.slice.call((c||d).querySelectorAll(p));
 72     return pa ? pa.call(this, r) : r;
 73   },
 74 /**
 75  * Returns a string representing the query source pattern.
 76  * @function {string} toString
 77  * @returns source pattern.
 78  */
 79   toString : function () {
 80     return this.pattern;
 81   },
 82 /**
 83  * Returns a string representing the source code of the Selector.
 84  * @function {string} toSource
 85  * @returns source code.
 86  */
 87   toSource : function () {
 88     return 'new Selector("'+this.pattern+'")';
 89   }

257   this.dup = this.srt = this.idx = this.i = this.nqp = 0;
258   with (this) {
259     var js = '';
260     do {
261       i = nqp = 0;
262       js += $('n=c;$1q:do{$2}while(false);', srt?'s=0;':'', type(qp, $(
263         srt?'for(x=r.length;s<x;z=s+((x-s)/2)|0,($1)?s=z+1:x=z);if(s<r.length)r.splice(s++,0,$2);else r[s++]=$2;':'r[s++]=$2;',
264         cdp?'r[z].compareDocumentPosition(n)&4':'h[r[z].LLn]<h[n.LLn]',
265         'pe?pe.call(this,n):n'

267     } while (qp = nqp);
268     js = $(
269       'var r=[],s=0,n,x,y,z,d=c?c.ownerDocument||c.document||c:c=document,pe=this.patchElement,pa=this.patchArray$1$2;$3return pa?pa.call(this,r):r;',
270       dup>0?',h={}':'',

279   type: function (qp, js, n, s, c) {
280     with (this) {
281       var m = /^\s*([\w-]+|\*)?(.*)/.exec(qp), t = m[1] || '*';
282       if (!n && c==' ' && !dup) dup = 1;
283       js = pred(m[2], js, n, t, c);
284       switch (c) {
285         case '>':
286           return cnlt && t!='*'
287                ? $('for(var n$1=n.children.tags("$2"),i$1=0;n=n$1[i$1++];){$3}', ++i, t, js)
288                : $(cnl ? 'for(var n$1=n.children,i$1=0;n=n$1[i$1++];)$2{$3}'
289                        : 'for(n=n.firstChild;n;n=n.nextSibling)$2{$3}', 
290                    ++i, t!='*'?'if(n.nodeName==="'+t.toUpperCase()+'")':!cnl||ie?'if(n.nodeType===1)':'', js);
291         case '+':
292           return $('while(n=n.nextSibling)if(n.node$1){$2break}else if(n.nodeType===1)break;',
293                    t=='*'?'Type===1':'Name==="'+t.toUpperCase()+'"', js);
294         case '~': 
295           return $('while(n=n.nextSibling)if(n.node$1){$3}else if(n.node$2)break;',
296                    t=='*'?'Type===1':'Name==="'+t.toUpperCase()+'"',
297                    s=='*'?'Type===1':'Name==="'+s.toUpperCase()+'"', js);
298         default:
299           return (typeof js == 'object') ? String(js) // handled by pred
300                : n ? t=='*' ? js : $('if(n.nodeName!="$1"){$2}', t.toUpperCase(), js)
301                : $('for(var n$1=n.getElementsByTagName("$2"),i$1=0;n=n$1[i$1++];)$3{$4}',

311            || /^:(nth)-(?:(last)-)?(?:(child)|of-type)\(\s*(?:(odd|even)|(-|\d*)n([+-]\d+)?|([1-9]\d*))\s*\)(.*)/.exec(qp)
