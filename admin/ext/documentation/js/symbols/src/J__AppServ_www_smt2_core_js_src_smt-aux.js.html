 30   /**
 31    * Allows recording/replaying the mouse path over Flash objects.
 32    * A Flash movie may display above all the layers on the HTML apge,
 33    * regardless of the stacking order ("z-index") of those layers.
 34    * Using a WMODE value of "opaque" or "transparent" will prevent a Flash movie from playing in the topmost layer
 35    * and allow you to adjust the layering of the movie within other layers of the HTML document.
 36    * However, to avoid possible performance issues, it's best use the "opaque" mode.
 37    * Note: The WMODE parameter is supported only on some browser/Flash Player version combinations.
 38    * If the WMODE parameter is not supported, the Flash movie will always display on top.
 39    * @return void
 40    */
 41   allowTrackingOnFlashObjects: function()
 42   {
 43     var obj = document.getElementsByTagName("object");
 44     for (var i = 0, t = obj.length; i < t; ++i) {
 45       var param = document.createElement("param");
 46       param.setAttribute("name", "wmode");
 47       param.setAttribute("value","opaque");
 48       obj[i].appendChild(param);
 49     }
 50 
 51     var embed = document.getElementsByTagName("embed");
 52     for (var j = 0, u = embed.length; j < u; ++j) {
 53       embed[j].setAttribute("wmode", "opaque");
 54       // recording on some browsers is tricky (replaying is ok, though)
 55       if (!/MSIE/i.test(navigator.userAgent)) {
 56         /* Some browsers sets the wmode correctly,
 57          * but once the SWF object is instantiated you can't update its properties.
 58          * So, replace the old Flash object with the new one ;)
 59          */
 60         var cloned = embed[j].cloneNode(true);
 61         embed[j].parentNode.replaceChild(cloned, embed[j]);
 62       }
 63     }
 64   },

 78   /**
 79    * Checks the DOM-ready initialization in modern browsers.
 80    * This method was introduced by Dean Edwards/Matthias Miller/John Resig (dean.edwards.name/outofhanwell.com/jquery.com)
 81    * and it is discussed in http://dean.edwards.name/weblog/2006/06/again/
 82    * It's 2009, and fortunately nowadays there are only 2 types of modern browsers: W3C standards and Internet Explorer.
 83    * @return void
 84    * @param {function} callback   the function to be called on DOM load
 85    */
 86   onDOMload: function(callback)
 87   {
 88     if (arguments.callee.done) { return; }
 89     arguments.callee.done = true;
 90 
 91     // Firefox, Opera, Webkit-based browsers (Chrome, Safari)...
 92     if (document.addEventListener) {
 93       document.addEventListener('DOMContentLoaded', callback, false);
 94     }
 95     // Internet Explorer ¬¬
 96     else if (document.attachEvent) {
 97       try {
 98         document.write("<scr"+"ipt id=__ie_onload defer=true src=//:><\/scr"+"ipt>");
 99         var script = document.getElementById("__ie_onload");
100         script.onreadystatechange = function() {
101           if (this.readyState === 'complete') { callback(); }
102         };
103       } catch(e) {}
104     }
105     else {
106       // fallback: old browsers use the window.onload event
107       this.addEvent(window, 'load', callback);
108     }

147    * Gets the position of element on page.
148    * @autor Peter-Paul Koch (quirksMode.org)
149    * @return {object} offset position - object with 2 properties: left {integer}, and top {integer}
150    */
151   findPos: function(obj)
152   {
153     var curleft = curtop = 0;
154     if (obj && obj.offsetParent) {
155       do {
156   			curleft += obj.offsetLeft;
157   			curtop  += obj.offsetTop;
158       } while (obj = obj.offsetParent);
159     }
160     
161     return { x:curleft, y:curtop };

257   /**
258    * Failsafe (and experimental) function.
259    * If your pages' HTML code have a lot of images and you do not allocate the image's dimensions (width/height attributes),
260    * then the page size sometimes could not be accurately computed and the tracking layer could be cropped.
261    * @return {object} viewport dimensions - object with 2 properties: width {integer}, and height {integer}
262    */
263   getHardcorePageSize: function()
264   {
265     // let's define some helpers
266     var w1=0, h1=0, w2=0, h2=0, w3=0, h3=0, w4=0, h4=0,
267         de = document.documentElement,
268         db = document.body;
269 
270     if (window.innerHeight) {
271       w1 = window.innerWidth;
272       h1 = window.innerHeight;
273       if (window.scrollMaxY) {
274         w1 += window.scrollMaxY;
275         h1 += window.scrollMaxY;
276       }
277     }
278     if (de && de.offsetHeight) {
279       w2 = de.offsetWidth;
280       h2 = de.offsetHeight;
281     }
282     if (db && db.scrollHeight) {
283       w3 = db.scrollWidth;
284       h3 = db.scrollHeight;
285     }
286     if (db && db.offsetWidth) {
287       w4 = db.offsetWidth;
288       h4 = db.offsetHeight;
289       if (db.offsetLeft) {
290         w4 += db.offsetLeft;
291         h4 += db.offsetTop;
292       }
293     }
294     // now return the max value
295     var maxHeight = Math.max(h1,h2,h3,h4);
296     var maxWidth = Math.max(w1,w2,w3,w4);
297 
298     return { width: maxWidth, height: maxHeight };
299   },

301   /**
302    * Gets the max z-index level available on the page.
303    * @return {integer}    z-index level
304    * @param {object} e    DOM element (default: document)
305    * @autor Jason J. Jaeger (greengeckodesign.com)
306    */
307   getNextHighestDepth: function(e)
308   {
309     var highestIndex = 0;
310     var currentIndex = 0;
311     var elementArray = [];
312     // check all elements in page ...
313     if (document.getElementsByTagName) {
314       elementArray = document.getElementsByTagName('*');
315     } else if (e.getElementsByTagName) {
316       elementArray = document.getElementsByTagName('*');
317     }
318     // ... and iterate
319     for (var i = 0, l = elementArray.length; i < l; ++i) {
320       if (elementArray[i].currentStyle) {
321         currentIndex = parseFloat(elementArray[i].currentStyle.zIndex);
322       } else if (window.getComputedStyle) {
323         currentIndex = parseFloat(document.defaultView.getComputedStyle(elementArray[i],null).getPropertyValue('z-index'));
324       }
325       if (currentIndex > highestIndex) { highestIndex = currentIndex; }
326     }
327 
328     return highestIndex + 1;
329   },

363   /**
364    * Rounds a number to a given digits accuracy.
365    * @return {float}
366    * @param {float}   number  input number
367    * @param {integer} digits  precision digits
368    */
369   roundTo: function(number,digits)
370   {
371     if (!digits) { digits = 2; }
372     var exp = 100; // faster, because smt2 precision is the same for all computations!
373     /* in 'taliban mode' that would be ok:
374      * <code>var exp = Math.pow(10,digits);</code>
375      * or even this, avoiding the pow function:
376      * <code>for (var i = 0, exp = 1; i < digits.length; ++i, exp *= 10) {}</code>
377      */
378     return Math.round(exp*number)/exp;

426   /**
427    * Makes an asynchronous XMLHTTP request (XHR) via GET or POST.
428    * Inspired on Peter-Paul Koch's XMLHttpRequest function.
429    * @return void
430    * @param  {object}    setup      Request properties
431    * @config {string}    url        Request URL
432    * @config {function} [callback]  Response function
433    * @config {string}   [postdata]  POST vars in the form "var1=name&var2=name..."
434    * @config {object}   [xmlhttp]   A previous XMLHTTP object can be reused
435    */
436   sendAjaxRequest: function(setup)
437   {
438     // create XHR object (or reuse it)
439     var request = (setup.xmlhttp) ? setup.xmlhttp : this.createXMLHTTPObject();
440     if (!request) { return; }
441 
442     var method = (setup.postdata) ? "POST" : "GET";
443     // start request
444     request.open(method, setup.url, true);
445     // post requests must set the correct content type
446     if (setup.postdata) {
447       request.setRequestHeader('Content-Type', "application/x-www-form-urlencoded");
448     }
449     // check for the 'complete' request state
450     request.onreadystatechange = function(){
451       if (request.readyState == 4 && typeof setup.callback === 'function') {
452         // send server response to callback function
453         setup.callback(request.responseText);
454       }
455     };
456     // send request
457     request.send(setup.postdata);
458   },

460   /**
461    * Cookies management object.
462    * This cookies object allows you to store and retrieve cookies easily.
463    * Cookies can be picked up by any other web pages in the correct domain.
464    * Cookies are set to expire after a certain length of time.
465    */
466   cookies: {
467     /**
468      * Stores a cookie variable.
469      * @return void
470      * @param {string} name
471      * @param {mixed}  value
472      * @param {string} expiredays (optional) default: no expire
473      * @param {string} domainpath (optional) default: root domain
474      */
475     setCookie: function(name,value,expiredays,domainpath)
476     {
477       var path = domainpath || "/";
478       var expires = "";
479       if (expiredays) {
480         var date = new Date();
481         date.setTime(date.getTime() + (expiredays*24*60*60*1000)); // ms
482         expires = "; expires=" + date.toGMTString();
483       }
484       document.cookie = name +"="+ escape(value) + expires +"; path=" + path;
485     },
486     /**
487      * Retrieves a cookie variable.
488      * @return {string}       cookie value, or false on failure
489      * @param {string} name   cookie name
490      */
491     getCookie: function(name)
492     {
493       var cStart,cEnd;
494       if (document.cookie.length > 0) {
495         cStart = document.cookie.indexOf(name+"=");
496         if (cStart != -1) {
497           cStart = cStart + name.length + 1;
498           cEnd   = document.cookie.indexOf(";", cStart);
499           if (cEnd == -1) {
500             cEnd = document.cookie.length;
501           }
502 
503           return unescape(document.cookie.substring(cStart, cEnd));
504         }
505       }
506       return false;
507     },
508     /**
509      * Checks if a cookie exists.
510      * @return {boolean}       true on success, or false on failure
511      * @param {string}  name   cookie name
512      */
513     checkCookie: function(name)
514     {
515       var c = this.getCookie(name);
516       return (c);
517     },
518     /**
519      * Deletes a cookie.
520      * @param {string}  name   cookie name
521      */
522     deleteCookie: function(name)
523     {
524       if (this.checkCookie(name)) {
525         this.setCookie(name, null, -1);
526       }
527     }

530   /**
531    * Core for tracking widgets.
532    * The word "widget" stands for *any* DOM element on the page.
533    * This snippet was developed years ago as 'DOM4CSS', and now lives in harmony with (smt).
534    */
535   widget: {
536     /**
537      * Finds the first available element with an ID.
538      * Traversing count starts from current element to node parents.
539      * This function should be registered on mouse move/down events.
540      * @return {object}            DOM node element
541      * @param {object}   e         DOM event
542      * @param {function} callback  response function
543      */
544     findDOMElement: function(e,callback)
545     {
546       if (!e) { e = window.event; }
547       // find the element
548       var t = e.target || e.srcElement;
549       // defeat Safari bug
550       if (t.nodeType == 3) { t = t.parentNode; }
551       // if the element has no ID, travese the DOM in reverse (find its parents)
552       var check = (t.id) ? this.getID(t) : this.getParents(t);
553       if (check) {
554         callback(check);
555       }
556     },
557     /**
558      * Gets the element's id.
559      * @return {string}     DOM node ID
560      * @param {object}  o   DOM node element
561      */
562     getID: function(o)
563     {
564       // do not track HTML and BODY nodes
565       if (o.nodeName == 'HTML' || o.nodeName == 'BODY') { return null; }
566       return o.nodeName + "#" + o.id;
567     },
568     /**
569      * Gets the element's class.
570      * If that element has more than one CSS class, return only the first class name found.
571      * @return {string}     DOM node class
572      * @param {object}  o   DOM node element
573      */
574     getClass: function(o)
575     {
576       // do not track HTML and BODY nodes
577       if (o.nodeName == 'HTML' || o.nodeName == 'BODY') { return null; }
578       // if the element has no class, return its node name only
579       return (o.className) ? o.nodeName + "." + o.className.split(" ")[0] : o.nodeName;
580     },
581     /**
582      * Gets all node parents until a node with ID is found.
583      * @return {string}     CSS selector string (intended to use in other scripts).
584      * @param {object}  o   DOM node element
585      */
586     getParents: function(o)
587     {
588       // store current node first
589       var elem = (o.id) ? this.getID(o) : this.getClass(o);
590       var list = [elem]; //(elem) ? [elem] : [];
591       // get all parents until find an element with ID
592       while (o.parentNode)
593       {
594         o = o.parentNode;
595         // store only element nodes
596         if (o.nodeType == 1)
597         {
598           if (o.id) {
599             elem = this.getID(o);
600             list.unshift(elem);
601             // if parent has an ID, end finding
602             return list.join(" ");
603           } else {
604             elem = this.getClass(o);
605             list.unshift(elem);
606           }
607           if (o == parent) {
608             // #document reached
609             return list.join(" ");
610           }
611         }
612       }
613       return list.join(" ");
614     }
615   },

617   /**
618    * Array methods -- without extending the Array prototype (best practice).
619    * Note: These Array methods would only work for a completely 'dense' array.
620    * If you're working with sparse arrays, then you should pre-process them:
621    * <code>
622    *   var narray = [];
623    *   for (var i = 0; i < array.length; ++i) {
624    *     if (array[i] != null) { narray.push(array[i]); }
625    *   }
626    *   // now narray is converted in a 'dense' array
627    * </code>
628    */
629   array: {
630     /**
631      * Gets the max element in a numeric array.
632      * @return {float}        array max value
633      * @param {array} array
634      * @author John Resig
635      */
636     max: function(arr)
637     {
638       return Math.max.apply(Math, arr);
639     },
640     /**
641      * Gets the min element in a numeric array.
642      * @return {float}        array min value
643      * @param {array} array
644      * @author John Resig
645      */
646     min: function(arr)
647     {
648       return Math.max.apply(Math, arr);
649     },
650     /**
651      * Sums all elements in a numeric array.
652      * @return {float}        array sum
653      * @param {array} arr
654      */
655     sum: function(arr)
656     {
657       var s = 0;
658       for (var i = 0, l = arr.length; i < l; ++i) {
659         s += arr[i];
660       }
661       return s;
662     },
663     /**
664      * Appends any supplied arguments to the end of the array, in the order given.
665      * @return {integer}        new length of the array
666      * @param {array} arr
667      * @param {mixed} content   anything to push to array
668      * @author Ash Searle (tweaked by Luis Leiva)
669      * @link http://hexmen.com/blog/
670      * @license http://creativecommons.org/licenses/by/2.5/
671      */
672     push: function(arr, content)
673     {
674       var args = Array.prototype.slice.call(content);
675       var n = arr.length >>> 0;
676       for (var i = 0; i < args.length; ++i) {
677         arr[n] = args[i];
678         n = n + 1 >>> 0;
679       }
680       arr.length = n;
681       return n;
682     },
683     /**
684      * Removes the last element of the array and returns it.
685      * @return {mixed}        element removed (if any). Otherwise return <code>undefined</code>
686      * @param {array} arr
687      * @author Ash Searle
688      * @link http://hexmen.com/blog/
689      * @license http://creativecommons.org/licenses/by/2.5/
690      */
691     pop: function(arr)
692     {
693       var n = arr.length >>> 0, value;
694       if (n) {
695         value = arr[--n];
696         delete arr[n];
697       }
698       arr.length = n;
699       return value;
700     },
701     /**
702      * Removes element(s) in desired position(s) from input array and returns the modified array.
703      * @return {array}          modified array
704      * @param {array}   arr
705      * @param {integer} from    start index
706      * @param {integer} to      end index
707      * @author John Resig
708      */
709     remove: function(arr, from, to)
710     {
711       var rest = arr.slice((to || from) + 1 || arr.length);
712       arr.length = (from < 0) ? arr.length + from : from;
713       return array.push.apply(arr, rest);
714 		},
715     /**
716      * Gets the index of the first element that matches search value.
717      * @return {integer}          index if found, or -1 otherwise
718      * @param {array}   arr
719      * @param {mixed}   search    element to search for
720      * @param {integer} from      start index
721      * @link http://snipplr.com/view/3355/arrayindexof/
722      */
723     indexOf: function(arr, search, from)
724     {
725       for (var i = (from || 0), total = arr.length; i < total; ++i) {
726         // strict types should use === comparison
727         if (arr[i] == search) {
728           return i;
729         }
730       }
731       return -1;
732     }
733   }
