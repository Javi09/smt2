253         smtRep.jg.drawPolyline([end.x-s,end.x,end.x+s], [end.y-s,end.y,end.y-s]);

278     /** 
279      * Draw time-depending circle.
280      */
281     drawVariableCircle: function(x,y, size) 
282     {
283       // use multiplier to normalize all circles: 0 < norm < 1
284       var norm = aux.roundTo(size/smtRep.jMax); 
285       if (size * norm === 0 ) { return; }
286       // limit size to 1/2 of current window width (px)
287       if (size > smtData.wcurr/2) { size = Math.round(smtData.wcurr/2 * norm); }
288       // draw
289       smtRep.jg.setColor(smtOpt.varCir);
290       smtRep.jg.drawEllipse(x - size/2, y - size/2, size, size);
291       smtRep.jg.paint();
292     },
293     /** 
294      * Gives a visual clue when the user is not using the mouse.

300       var fontSize   = 16,
301           circleSize = 50;
302       smtRep.jgHelper.setColor(smtOpt.varCir);
303       smtRep.jgHelper.fillEllipse(x - circleSize/2, y - circleSize/2, circleSize, circleSize);
304       smtRep.jgHelper.setColor("black");
305       smtRep.jgHelper.setFont("sans-serif", fontSize+'px', Font.BOLD);
306       // center the text in vertical 
307       smtRep.jgHelper.drawString("stopped...", x, y - fontSize/2);
308       smtRep.jgHelper.paint();
309     },   
310     /** 
311      * Draw centroid as a star.
312      */
313     drawCentroid: function()
314     {
315       smtRep.jg.setColor(smtOpt.cenPt);
316       // the centroid is computed discarding null distances
317       var u = Math.round(aux.array.sum(xclean) * smtRep.discrepance.x / xclean.length),
318           v = Math.round(aux.array.sum(yclean) * smtRep.discrepance.y / yclean.length),
319           l = 20; // centroid line length
320       smtRep.jg.setStroke(5);
321       smtRep.jg.drawLine(u, v, u+l, v-l); // 1st quadrant
322     	smtRep.jg.drawLine(u, v, u-l, v-l); // 2nd quadrant
323     	smtRep.jg.drawLine(u, v, u-l, v+l); // 3rd quadrant
324     	smtRep.jg.drawLine(u, v, u+l, v+l); // 4th quadrant
325     	smtRep.jg.setStroke(0); // reset strokes
326     	smtRep.jg.paint();
327     },
328     /** 
329      * Draw cluster as a circle.
330      */
331     drawClusters: function(response) 
332     {
333       var K = JSON.parse(response);
334       // again (same as in Flash) typeof K is string, so re-parse
335       if (typeof K !== 'object') {
336         K = JSON.parse(K);
337       }
338       smtRep.jgClust.setColor(smtOpt.clust);
339       for (var i = 0, size = 0, numClusters = K.sizes.length; i < numClusters; ++i) {
340         size = K.sizes[i];
341         smtRep.jgClust.fillEllipse(K.xclusters[i] * smtRep.discrepance.x - size/2, K.yclusters[i] * smtRep.discrepance.y - size/2, size, size);
342       }
343       smtRep.jgClust.paint();
344     },
345     /** 
346      * Get euclidean distance from point a to point b.
347      */
348     distance: function(a,b) 
349     {
350       return Math.sqrt( Math.pow(a.x - b.x,2) + Math.pow(a.y - b.y,2) );
351     },
352     /** 
353      * Auto scrolls the browser window.
354      */
355     checkAutoScrolling: function(x,y) 
356     {
357       if (!smtOpt.realTime) { return; }
358       // center current mouse coords on the viewport
359       aux.doScroll({xpos:x, ypos:y, width:smtRep.viewport.width, height:smtRep.viewport.height});
360     },
361     /** 
362      * (smt)2 realtime drawing algorithm.
363      */
364     playMouse: function() 
365     {
366       if (smtRep.paused) { return; }
367 
368       // mouse coords normalization
369       var iniMouse = { 
370                         x: user.xcoords[smtRep.i] * smtRep.discrepance.x,
371                         y: user.ycoords[smtRep.i] * smtRep.discrepance.y 
372                      };
373       var endMouse = { 
374                         x: user.xcoords[smtRep.i+1] * smtRep.discrepance.x,
375                         y: user.ycoords[smtRep.i+1] * smtRep.discrepance.y 
376                      };
377       var iniClick = {
378                         x: user.xclicks[smtRep.i] * smtRep.discrepance.x, 
379                         y: user.yclicks[smtRep.i] * smtRep.discrepance.y
380                      };
381       var endClick = {
382                         x: user.xclicks[smtRep.i+1] * smtRep.discrepance.x, 
383                         y: user.yclicks[smtRep.i+1] * smtRep.discrepance.y
384                      };
385       
386       // draw entry point
387       if (smtRep.i === 0) {
388         smtRep.drawCursor(iniMouse.x,iniMouse.y, smtOpt.entryPt);
389       }
390       
391       // main loop to draw mouse trail
392       if (smtRep.i < user.xcoords.length) 
393       {
394         var mouseDistance = smtRep.distance(iniMouse,endMouse);
395         // draw registration points
396         if (mouseDistance) {
397           // there is mouse movement
398           if (!smtOpt.dirVect) {
399             // show static squares
400             smtRep.drawRegistrationPoint(iniMouse.x,iniMouse.y);
401           } else {
402             // show direction pseudo-arrows
403             smtRep.drawDirectionArrow(iniMouse,endMouse);
404           }
405           // variable circles
406           if (smtRep.j > 1) {
407             smtRep.drawVariableCircle(iniMouse.x, iniMouse.y, smtRep.j);
408             smtRep.jgHelper.clear();
409           }
410           // reset variable circles size
411           smtRep.j = 1;
412         } else {
413           // mouse stop: store variable size (circles)
414           ++smtRep.j;
415           smtRep.drawMouseStop(iniMouse.x, iniMouse.y);
416         }
417         // draw lines
418         smtRep.drawLine(iniMouse,endMouse);
419         // draw clicks
420         if (iniClick.x) {
421           var clickDistance = smtRep.distance(iniClick,endClick);
422           if (!clickDistance) {
423             // is a single click
424             smtRep.drawClick(endClick.x,endClick.y, false);
425           } else if (endClick.x !== 0) {
426             // is drag and drop
427             smtRep.drawClick(iniClick.x,iniClick.y, true);
428           }
429         }
430         // update mouse coordinates
431         ++smtRep.i;
432         // check auto scrolling
433         smtRep.checkAutoScrolling(endMouse.x, endMouse.y);
434     	}
435       
436       // draw exit point
437       else {
438     	  // rewind count 1 step to access the last mouse coordinate
439     	  --smtRep.i;
440     	  iniMouse.x = user.xcoords[smtRep.i] * smtRep.discrepance.x;
441         iniMouse.y = user.ycoords[smtRep.i] * smtRep.discrepance.y;
442         // draw exit point
443     	  smtRep.drawCursor(iniMouse.x,iniMouse.y, smtOpt.exitPt);
444     	  // draw clusters
445     	  var data  = "xhr=1"; 
446             data += "&xdata=" + JSON.stringify(user.xcoords);
447             data += "&ydata=" + JSON.stringify(user.ycoords);
448         
449         var basepath = aux.getBase();
450         // send request
451         aux.sendAjaxRequest({
452           url:       basepath + "includes/kmeans.php", 
453           postdata:  data,
454           callback:  smtRep.drawClusters
455         });
456         
457         // draw centroid (average mouse position) 
458         smtRep.drawCentroid();
459         // clear mouse tracking
460         clearInterval(smtRep.play);
461         smtRep.play = null;
462         smtRep.jgHelper.clear();
463         if (numUsers == 1) {
464           // load next trail
465           aux.loadNextMouseTrail(smtData);
466         }
467     	}
468     },
469     /** 
470      * Replay method: static or dynamic.
471      */
472     replay: function(realtime) 
473     {
474       if (realtime) {
475         // fps are stored in smtData object, so we can use that value here
476         var interval = Math.round(1000/smtData.fps);
477         smtRep.play = setInterval(smtRep.playMouse, interval);
478       } else {
479         // static mouse tracking visualization 
480         for (var k = 0, total = user.xcoords.length; k <= total; ++k) {
481           smtRep.playMouse();
482         }
483       }
484     },
485     /** 
486      * Reload method: mouse tracking layers are redrawn.
487      */
488     reset: function() 
489     {
490       clearInterval(smtRep.play);
491       smtRep.paused = false;
492       // reset counters
493       smtRep.i = 0;
494       smtRep.j = 1;    
495       // clear canvas  
496       smtRep.jg.clear();
497       smtRep.jgClust.clear();
498     },
499     /** 
500      * User can pause the mouse replay by pressing the SPACE key, 
501      * or toggle replay mode by pressing the ESC key.
502      */
503     helpKeys: function(e) 
504     {
505       // use helpKeys only in realtime replaying
506       if (!smtOpt.realTime) { return; }
507       
508       if (!e) { e = window.event; }
509       var code = e.keyCode || e.which;
510       // on press ESC key finish drawing
511       if (code == 27) {
512         // clear main loop
513         clearInterval(smtRep.play);
514         smtRep.paused = false;
515         // set this flag
516         smtOpt.realTime = false;
517         // end drawing from the current position
518         for (var k = smtRep.i, total = user.xcoords.length; k <= total; ++k) {
519           smtRep.playMouse();
520         }
521       } else if (code == 32) {
522         // on press space bar toggle drawing
523         smtRep.paused = !smtRep.paused;
524       }
525     },
526     /** 
527      * System initialization.
528      */
529     init: function() 
530     {
531       // use vieport size to auto-scroll window
532       var vp = aux.getWindowSize();
533       smtRep.viewport.width = vp.width;
534       smtRep.viewport.height = vp.height;
535       // normalize data
536       var doc = aux.getPageSize();
537       smtRep.page.width = doc.width;
538       smtRep.page.height = doc.height;
539       // compute the discrepance ratio
540       if (user.wprev && user.hprev) {
541         smtRep.discrepance.x = aux.roundTo(doc.width / user.wprev);
542         smtRep.discrepance.y = aux.roundTo(doc.height / user.hprev);
543       }
544           
545       // precalculate the user stops: useful for time-depending circles and path centroid
546       var stops = [];      
547       var size = 1;
548       for (var k = 0, len = user.xcoords.length; k < len; ++k) {
549         if (user.xcoords[k] == user.xcoords[k+1] && user.ycoords[k] == user.ycoords[k+1]) {
550           ++size;
551         } else {
552           // store all user stops (time) for drawing variable circles later
553           if (size > 1) { stops.push(size); }
554           size = 1;
555           // store clean mouse coordinates
556           xclean.push(user.xcoords[k]);
557           yclean.push(user.ycoords[k]);
558         }
559       }
560       // set max size for variable circles
561       smtRep.jMax = aux.array.max(stops);       
562       // common suffix for tracking canvas and background layers
563       var smtName = "smtCanvas";
564       // set the canvas layer
565       smtRep.createCanvas(smtName);
566       // draw the background layer
567       if (smtOpt.bgLayer) { smtRep.setBgCanvas(smtName); }
568       // allow mouse replay over Flash animations
569       aux.allowTrackingOnFlashObjects();
570       // init
571       smtRep.replay(smtOpt.realTime);
572     }
573     
574   };
575   
576   // do not overwrite the smt2 namespace
577   if (typeof window.smt2 !== 'undefined') { throw("smt2 namespace conflict"); }
578   // else expose replay method
579   window.smt2 = {
580     replay: function(opts) {
581       // load custom smtOpt, if set
582       if (typeof opts !== "undefined") { aux.overrideTrackingOptions(smtOpt, opts); }
583       // replay
584       aux.addEvent(document, "keyup",  smtRep.helpKeys);
585       //aux.addEvent(window, "resize", smtRep.reset);
586       //aux.addEvent(window, "resize", aux.reloadPage);
587       aux.onDOMload(smtRep.init);
588     }
589   }
