  318.          * @return boolean True if the browser is the BlackBerry browser otherwise false
  319.          */
  320.         protected function checkBrowserBlackBerry() {
  321.             $retval = false;
  322.             if( preg_match('/blackberry/i',$this->_agent) ) {
  323.                 $aresult = explode("/",stristr($this->_agent,"BlackBerry"));
  324.                 $aversion = explode(' ',$aresult[1]);
  325.                 $this->setVersion($aversion[0]);
  326.                 $this->_browser_name = self::BROWSER_BLACKBERRY;
  327.                 $this->setMobile();
  328.                 $retval = true;
  329.             }
  330.             return $retval;
  331.         }

  336.          */
  337.         protected function checkForAol() {
  338.             $retval = false;
  339.             if( preg_match('/aol/i', $this->_agent) ) {
  340.                 $aversion = explode(' ',stristr($this->_agent, 'AOL'));
  341.                 $this->setAol(true);
  342.                 $this->setAolVersion(preg_replace('/[^0-9\.a-z]/i', '', $aversion[1]));
  343.                 $retval = true;
  344.             }
  345.             else {
  346.                 $this->setAol(false);
  347.                 $this->setAolVersion(self::VERSION_UNKNOWN);
  348.                 $retval = true;
  349.             }
  350.             return $retval;
