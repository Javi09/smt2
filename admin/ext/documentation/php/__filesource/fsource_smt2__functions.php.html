  268. /**

  273.  * @param   array  $getDistances   if TRUE, the result array contains euclidean distances

  276. function convert_points($xcoords, $ycoords, $getDistances = false) 
  277. {
  278.   // initialize points array
  279.   $pointArray = array();
  280.   // check for illegal offsets on $coords

  283.   foreach ($xcoords as $i => $value) 
  284.   {
  285.     $p = new Point($value, $ycoords[$i]);

  294.     } else {
  295.       // append point to the points array, discarding null distances
  296.       if ($distance > 0) { $pointArray[] = $p; }
  297.     }
  298.   }

  300.   return $pointArray;
