  180.   protected function vectorNorm($array, $type)
  181.   {
  182.     /* Note: mouse coordinates are always positive values, so we don't need
  183.      * to compute the absolute values first for cases "L1", "Linf" and "Linf2".
  184.      */
  185.     switch ($type)
  186.     {
  187.       /** 1-norm */
  188.       case "L1":
  189.         $sum = 0;
  190.         foreach ($array as $value) { $sum += $value; }
  191.         $result = $sum;
  192.         break;
  193.  
  194.       /** Unscaled 2-norm */
  195.       case "uL2":
  196.       default:
  197.         $sum = 0;
  198.         foreach ($array as $value) { $sum += $value*$value; }
  199.         $result = sqrt($sum);
  200.         break;
  201.  
  202.       /** Scaled 2-norm */
  203.       case "sL2":
  204.         $points = array();
  205.         foreach ($array as $value) { $points[] = $value; }
  206.         $rmax = max($points);
  207.         $sum = 0;
  208.         foreach ($array as $value) {
  209.           $value /= $rmax;
  210.           $sum += $value * $value;
  211.         }
  212.         $result = $rmax * sqrt($sum);
  213.  
  214.         break;
  215.  
  216.       /** Infinity norm */
  217.       case "Linf":
  218.         $points = array();
  219.         foreach ($array as $value) { $points[] = $value; }
  220.         $result = max($points);
  221.         break;
  222.  
  223.       /** Negative infinity norm */
  224.       case "Linf2":
  225.         $points = array();
  226.         foreach ($array as $value) { $points[] = $value; }
  227.         $result = min($points);
  228.         break;
  229.     }
  230.  
  231.     return $result;
  232.   }
